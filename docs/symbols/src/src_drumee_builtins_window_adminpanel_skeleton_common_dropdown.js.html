  7 const __skl_admin_panel_common_dropdown = function(_ui_) {
  8   if (!_ui_.organisation.id){
  9     return '';
 10   }
 11   let pages = [
 12     { display:'Domain',       value :'domain_page'},
 13     { display:'Members',      value:'members_page'},
 14     { display:'Import a list',value:'import_list_page'},
 15     { display:'Security',     value:'security_page'},
 16     { display:'Subscription', value:'subscription_page'}
 17   ];
 18   
 19 
 20   const dropdownMenu = {
 21     kind: 'widget_dropdown_menu',
 22     options : pages
 23   }
 24   
 25   let a = Skeletons.Box.Y({
 26     className  : `${_ui_.fig.family}__dropdown_main`,
 27     debug      : __filename,
 28     kids       : [
 29       Skeletons.Box.X({
 30       className  : `${_ui_.fig.family}__dropdown_container`,
 31       kids : [dropdownMenu]
 32       })
 33     ]
 34   })
 35   return a;
 36 };
 37 
 38 export default __skl_admin_panel_common_dropdown;
