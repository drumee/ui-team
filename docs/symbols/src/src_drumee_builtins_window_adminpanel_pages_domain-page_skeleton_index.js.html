 10 function __skl_domain_page (_ui_) {
 11   let footer= ''
 12   let validate_button = Skeletons.Note({
 13     className   : `${_ui_.fig.family}__button_val input_btn`,
 14     service     : 'added_domain',
 15     content     : LOCALE.VALIDATE //"Validate"
 16   })
 17 
 18   if(_ui_.mode == _a.new)
 19     footer =  validate_button
 20   else if(_ui_.mode == _a.view) {
 21     footer = Skeletons.Note({
 22       className   : `${_ui_.fig.family}__button_val input_btn`,
 23       service     : 'edit_domain',
 24       content     : "Edit",
 25       uiHandler : [_ui_]
 26     })
 27   } else if(_ui_.mode == _a.edit) {
 28     footer = Preset.ConfirmButtons(_ui_, {
 29       cancelLabel     : LOCALE.CANCEL || '',
 30       cancelService   : 'view_domain',
 31       confirmLabel    : 'Update' || '',
 32       confirmService  : 'update_domain'
 33     })
 34   }
 35 
 36 
 37   let a = Skeletons.Box.Y({
 38     className  : `${_ui_.fig.family}__main`,
 39     debug      : __filename,
 40     kids       : [
 41       Skeletons.Box.Y({
 42         className  : `${_ui_.fig.family}__container`,
 43         kids : [
 44           Skeletons.Box.X({
 45             className  : `${_ui_.fig.family}__body`,
 46             kids: [
 47               require('./domain').default(_ui_)
 48             ]
 49           }),
 50           Skeletons.Box.X({
 51             className  : `${_ui_.fig.family}__footer`,
 52             kids: [
 53               footer
 54             ]
 55           })
 56         ]
 57       })
 58     ]
 59   })
 60   return a;
 61 }
 62 export default __skl_domain_page;
