 58    */
 59   onUiEvent (cmd, args) {
 60     const service = cmd.get(_a.service) || cmd.get(_a.name);
 61     this.debug('onUiEvent', service, cmd, this);
 62 
 63     switch(service) {
 64       case 'show-member-list':
 65         this.loadMembersList(cmd.source)
 66         break;
 67       
 68       case 'show-member-detail':
 69         this.loadMemberDetail(cmd.source)
 70         break;
 71       
 72       case 'create-member':
 73         this.loadCreateMember()
 74         break
 75       
 76       case 'cancel-create-member':
 77         this.cancelCreateMemberForm()
 78         break
 79       
 80       case 'add-member':
 81         this.addNewMember(cmd)
 82         break     
 83       
 84       case _e.edit:
 85         this.loadEditMember(cmd)
 86         break
 87       
 88       case 'update-member':
 89         this.updateMember(cmd)
 90         break
 91       
 92       case _e.delete:
 93         this.deleteMember(cmd)
 94         break
 95       
 96       case 'confirm-delete':
 97         this.confirmDeleteMember()
 98         break
 99       
100       case 'choose-admins':
101         this.loadChooseAdminLayer()
102         break
103       
104       case 'previous-page':
105         this.previousBreadcrumb()
106         break
107       
108       case 'next-page':
109         this.nextBreadcrumb()
110         break
111       
112       case 'close-overlay':
113         this.closeOverlay(cmd)
114         break
115       
116       default:
117         this.debug("Created by kind builder 222");
118     }
119   }

239     this.debug('deleteMember', cmd, this)
240     const parent = this.mget(_a.parent)
241     const overlayWrapper = this.getPart('overlay-wrapper')
242     overlayWrapper.el.dataset.mode = _a.open
243     this.mset(_a.member, cmd.mget(_a.member))

248   /**
249    * 
250   */
251   confirmDeleteMember () {
252     this.debug('confirmDeleteMember', this)
253     const memberData = this.mget(_a.member)
254     if (memberData && memberData.drumate_id) {
255       return this.deleteAcknowledgement()  //to be called from the _dispatchRest
256     }
257   } 
258 
259   /* 
260    *
261   */
262   deleteAcknowledgement () {
263     const memberData = this.mget(_a.member)
264     let memberList = this.getItemsByKind('widget_members_list')[0]
265     const deleteAck = this.getPart('delete-confirm-content')
266     
267     deleteAck.feed(require('./skeleton/delete-member/delete-acknowledgement').default(this))
268 
269     memberList.deleteMemberItemById(memberData.drumate_id, this)
270     const f = () => {
271       return memberList.triggerClick()
272     }
273     
274     return _.delay(f, 500)
275   }

277   /**
278    * 
279   */
280   previousBreadcrumb () {
281     let viewInstance = this.viewInstance || 1
282     if (viewInstance > 1)
283       viewInstance = viewInstance - 1
284     return this.updateInstance(viewInstance)
285   }
286 
287   /**
288    * 
289   */
290   nextBreadcrumb () {
291     let viewInstance = this.viewInstance || 1
292     if (viewInstance < 3)
293       viewInstance = viewInstance + 1
294     return this.updateInstance(viewInstance)
295   }

297   /**
