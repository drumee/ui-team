 87   seek_insertion(moving){
 88     let file;
 89     try{
 90       file = moving.event.originalEvent.dataTransfer.items[0];
 91     }catch(e){
 92       this.warn("INVALID FILE");
 93     }
 94     if(file && /excel/.test(file.type)){
 95       this.debug("HANDLE VALID FROM HERE",file);
 96     }else{
 97       this.debug("HANDLE INVALID FILE FROM HERE",file);
 98     }
 99     return this;

103   /**

105   insertMedia(f){
106     if(f.length !=1){
107       Wm.alert(LOCALE.ONLY_ONE_FILE_ALLOWED);
108       return;
109     }
110     let file = f[0].get(_a.file);
111     if(file.isDirectory || !/(.+)\.xls(.*)$/.test(file.name)){
112       this.warn("REJECTED ", file, file.isDirectory, file.name, /(.+)\.xls*$/.test(file.name));
113       Wm.alert(LOCALE.REQUIRES_FILE_TYPE.format('excel'.underline("required-type")));
114       return;
115     }
116 
117     let opt = {
118       file     : file,
119       filename : file.name,
120       filesize : file.size,
121       service  : _SVC.adminpanel.members_import,
122       hub_id   : Visitor.id, 
123       nid      : '0' // Only to avoid uloader complain
124     }
125     this.$el.append(__progess(this));
126     let id = `${this._id}-fg`;
127     Utils.waitForEl(id, ()=>{
128       this.$progress = $(`#${id}`);
129       let socket = new __uploader(opt);
130       socket.addListener(this);
131       socket.send(_K.url.service);
132     });
133   }

183  /* *********************************************************
184   * getMyOrg 
185   * ********************************************************* */
186   getMyOrg(callback){
187     const data = {
188       service: _SVC.adminpanel.my_organisation,
189       hub_id  : Visitor.id
190     }
191     this.fetchService(data, {async : 0})
192     .then((data)=>{
193       this.organisation = data;
194       this.updateDropdown();      
195       this.updateTitle();
196       callback && callback();
197       this.debug('_SVC.adminpanel.my_organisation',_SVC.adminpanel.my_organisation,data,this);
198    }).catch(error => this.warn(error));
199   }

201  /* *********************************************************
202   * Router
203   * ********************************************************* */
204   router(page = 'domain_page', options = {}) {
205     let route = () => {
206      // page = 'import_list_page'
207       switch (page) {
208         case 'domain_page':
209           this.loadDomainPage();
210           break;
211         case 'members_page':
212           this.loadMembersPage();
213           break;
214         case 'import_list_page':
215           this.loadImportListPage();
216           break;
217         case 'security_page':
218           this.loadSecurityPage();
219           break;
220         case 'subscription_page':
221         default:
222           this.debug("Created by kind builder");
223           this.loadingPage()
224       }
225     }
226 
227     if (this.organisation && this.organisation.id && this.__content) {
228       route();
229       return;
230     }
231    this.getMyOrg(route)
232   }
