142   canNegociate() {
143     if(!this._origin.stream_id){
144       this.warn("Attempt to negociate without local stream_id");
145       return false;
146     }
147     if(!this._target.stream_id){
148       this.warn("Attempt to negociate without remote stream_id");
149       return false;
150     }
151     if(this._origin.stream_id == this._target.stream_id){
152       this.warn("Attempt to negociate with same stream");
153       return false;
154     }
155     return true;
156   }

277   async receiveOffer(data) {
278     //if(!this.canNegociate()) return;
279     this.debug("HHHHUUU receiveOffer", data.target, this.origin());
280     const offer = data.offer;
281     if(_.isEmpty(offer)){
282       this.warn('No effor received.');
283       return;
284     }
285     //this.debug('Received OFFER: ', offer.sdp);
286     const pc = this.pc;
287     let desc = await new RTCSessionDescription(offer);
288     await pc.setRemoteDescription(desc);
289     await this.addLocalStream();
290     const answer = await pc.createAnswer();
291     await pc.setLocalDescription(answer);
292     // this.debug('Create ANSWER: ', answer, this);
293     const opt = {
294       service   : _SVC.signaling.have_answer,
295       hub_id    : this.mget(_a.hub_id),
296       answer,
297       origin    : this.origin(),
298       target    : this.target(),
299     };
300     let s = new Service(opt);
301     await s.postData();
302     //this.debug(">>HHHH ANSER sent", opt, answer.sdp);
303     return answer;
304   }

356   send(args) {
357     let target = this.target();
358     if(_.isEmpty(target)){
359       this.warn(_ERR.attr_required.format(_a.target));
360       return;
361     }
362     let scope = _a.hub;
363     if(target.socket_id){
364       scope = _a.socket;
365     }else if(target.uid){
366       scope = _a.user;
367     }
368     let opt = {
369       service : _SVC.signaling.message,
370       hub_id  : this.get(_a.hub_id),
371       audio   : this.mget(_a.audio),
372       video   : this.mget(_a.video),
373       scope,
374       origin  : this.origin(),
375       target  : this.target(),
376       ...args,
377     };
378     ////this.debug("HHHHHH SENDING", opt);
379     this.postData(opt);
380   }
