 40   _onconnectionstatechange(e){
 41     this.debug(`XWQ SIG == Connection state change: ${this.localPC.connectionState}`);
 42     switch(this.localPC.connectionState){
 43       case 'connected':
 44         if (this.localPC.signalingState == 'stable'){
 45           this.sendSignaling(_e.enter);
 46         }else{
 47           this._onStable=()=>{
 48             this.sendSignaling(_e.enter);
 49             this._onStable=null;
 50           }
 51         }
 52         break;
 53       case  'disconnected':
 54         this.debug(`XWQ ICE == DISCONNECTED: `, e);
 55       }
 56   }
 57 
 58   // ===========================================================
 59   //
 60   // ===========================================================
 61   _onsignalingstatechange(e){
 62     this.debug(`XWQ SIG == Signaling state change: ${this.localPC.signalingState}`, e);
 63     //this.isReady = true;
 64     this.isNegotiating = (this.localPC.signalingState != "stable");
 65     if (this.localPC.signalingState == 'stable'){
 66       try{
 67         this._onStable();
 68       }catch(err){
 69         // Never mind
 70         // this.warn("GOT ERROR", err);
 71       }
 72     }
 73   }

 78   async _onnegotiationneeded(e){
 79     this.debug("XWQ ===== In negociation 78",  this.localStream.id, e, this);
 80     // if(this.isNegotiating || !this.isReady){
 81     if(this.isNegotiating){
 82         this.debug("XWQ  In negociation");
 83       return;
 84     }
 85     this.isNegotiating = true;
 86     try {
 87       let offer = await this.localPC.createOffer();
 88       await this.localPC.setLocalDescription(offer);
 89       const opt = {
 90         service   : _SVC.signaling.have_offer,
 91         hub_id    : this.mget(_a.hub_id),
 92         mode      : this.mget(_a.mode),
 93         key       : Visitor.get(_a.socket_id),
 94         socket_id : this.mget(_a.socket_id),
 95         offer
 96       };
 97       let s = new Service(opt);
 98       await s.postData();
 99       this.debug("XWQ ===== In negociation 87", opt, this.localStream.id, e, this);
100       return offer;
101     } catch (e) {
102       this.warn("FAILED TO HANDL negociation",e);  
103     }
104     return null;
105   }

111   _ontrack(e) {
112     const stream = e.streams[0];
113     if(this.streams.get(stream.id)) return;
114     this.streams.set(stream.id, stream);
115     this.debug('114 ZZZ XWQ Got remote track:', this, e, stream);
116     this.triggerMethod('add:stream',{
117       type: 'remote', 
118       stream, 
119       label : LOCALE.CONNECTING
120     })
121   }

126   _onicecandidate(e){
127     this.debug(
128       `FFFF XWWQ ON ONICECANDIDATE: ${this.localPC.iceConnectionState} \
129       ${this.localPC.signalingState}`, e);
130 
131     if(e.candidate){
132       const opt = {
133         service   : _SVC.signaling.add_candidate,
134         candidate : e.candidate,
135         hub_id    : this.mget(_a.hub_id),
136         mode      : this.mget(_a.mode),
137         key       : Visitor.get(_a.socket_id),
138         socket_id : this.mget(_a.socket_id),
139       };
140       this.postData(opt);
141     }
142   }

144   async handleCandidate(candidate, dest, type) {
145     let desc = new RTCIceCandidate(candidate);
146     this.debug(`XWQ HHHH type=${type} : ${dest.iceGatheringState}`, candidate, desc);
147     try{
148       await dest.addIceCandidate(candidate);
149     }catch(err){
150       this.debug(`XWQ HHHH GOT ERROR`,  err);
151     }
152   }

157   setupConnection() {
158     const {configuration} = require('./configs');
159     this.localPC = new RTCPeerConnection(configuration);
160     this.remotePC = new RTCPeerConnection(configuration);
161 
162     this.isNegotiating = false;
163     this.localPC.onicegatheringstatechange = (e) => {
164       this.debug(
165         `XWQ SEE ICE gathering LOCAL: ${this.localPC.iceGatheringState}`, e);
166         this.debug(
167           `XWQ SEE ICE gathering REMOTE: ${this.remotePC.iceGatheringState}`, e);
168       };
169     this.localPC.oniceconnectionstatechange = (e) => {
170       this.debug(
171         `XWQ SEE ICE connection LOCAL : ${this.localPC.iceConnectionState}`, e);
172         this.debug(
173           `XWQ SEE ICE connection REMOTE : ${this.remotePC.iceConnectionState}`, e);
174       };
175   
176     this.localPC.onconnectionstatechange = this._onconnectionstatechange.bind(this);
177     this.localPC.onsignalingstatechange  = this._onsignalingstatechange.bind(this);
178     this.localPC.onnegotiationneeded = this._onnegotiationneeded.bind(this);  
179     this.remotePC.ontrack = this._ontrack.bind(this);
180 
181     this.localPC.onicecandidate =  (e) => {
182       this.handleCandidate(e.candidate, this.remotePC, 'local');
183     };
184 
185     this.remotePC.onicecandidate =  (e) => {
186       this.handleCandidate(e.candidate, this.localPC, 'remote');
187     };
188     // this._onicecandidate.bind(this);
189 
190   }

196   setupTracks(type, state) {
197     let stream = this.localStream;
198     let tracks = [];
199     switch(type){
200       case _a.video:
201         tracks = stream.getVideoTracks();
202         this.mset(type, state);
203         break;
204       case _a.audio:
205         tracks = stream.getAudioTracks();
206         this.mset(type, state);
207         break;
208     }
209     if(!tracks.length){
210       return;
211     }
212     for(let t of tracks){
213       t.enabled = Utils.boolean(state);
214     }
215   }

220   onUiEvent(cmd, args) {
221     let service = cmd.mget(_a.service);
222     let state   = cmd.mget(_a.state)
223     switch(service){
224       case _a.video: case _a.audio:
225         this.setupTracks(service, state);
226         this.sendSignaling(_e.update);
227     }

230   /* 
231   * To inform others who is behind the particular localstream
232   * 
233   * ----------------------------------------------------------- */
234   sendSignaling(event) {
235     if(!this.localStream.id){
236       return;
237     }
238     this.postData({
239       service     : _SVC.signaling.update,
240       key         : Visitor.get(_a.socket_id),
241       hub_id      : this.mget(_a.hub_id),
242       socket_id   : this.mget(_a.socket_id),
243       stream_id   : this.localStream.id,
244       mode        : this.mget(_a.mode),
245       event       : event,
246       video       : this.mget(_a.video),
247       audio       : this.mget(_a.audio),
248     })
249   }

273   probeDevices() {
274     const self =  this;
275     //self.debug("SELECT JJJJJJJ");
276     if (!navigator.mediaDevices) {
277       this._noDevice("No devices found").bind(this);
278       return;
279     }
280 
281     navigator.mediaDevices.enumerateDevices()
282     .then((devices)=>{
283       this.devices = devices;
284       this.triggerHandlers({
285         service: "devices-list", 
286         devices:devices
287       });
288     })
289     .catch(this._noDevice.bind(this));
290   }

295   _noDevice(err){
296     this.warn("FAILED TO GET USER DEVICE", err);
297     window.__uu = err;
298     let msg = err;
299     if(err != null && err.message){
300       switch(err.name){
301         case "NotReadableError":
302           msg = `${err.message}. It's likely already in use.`;
303           break;
304         default:
305           msg = `${err.message}. Your browser doesn't support video feature.`;
306         }
307       this.feed(
308         Skeletons.Note(msg, `${this.fig.family}__no-device`)
309       );
310     }
311   };

331   /* ------------------------------------------------------------
332   * To request permission to access user media (micro and/or video)
333   * @params {object} {audio:boolean, video:boolean}
334   * @retturn stream bound to granted media
335   * ------------------------------------------------------------ */
336   async requestUserMedia() {
337     const constraints = require('./configs').constraints();
338     this.debug("XWQ requestUserMedia", constraints, this);
339     let  stream;
340     if (navigator.mediaDevices) {
341       stream = await navigator.mediaDevices.getUserMedia(constraints);
342       this.localStream = stream;
343       this.streams.set(stream.id, stream);
344       this.setupTracks(_a.video, this.mget(_a.video));
345       this.setupTracks(_a.audio, this.mget(_a.audio));
346   } else { 
347       Cop.say(`mediaDevices unsupported`);
348       return null;
349     }
350     return stream;
351   }

353   /* --------------------------------------------------
354    * 
355    * @params {object} offer sent by caller
356    * -------------------------------------------------- */
357   async receiveOffer(data) {
358     const offer = data.offer;
359     if(_.isEmpty(offer)){
360       this.warn('No effor received.');
361       return;
362     }
363     this.debug('Received OFFER: ', data, this);
364     const pc = this.localPC;
365     let desc = await new RTCSessionDescription(offer);
366     this.debug(`XWQ TT SIG 344 == \
367       ${pc.signalingState} CONN=${pc.connectionState} NEG=${this.isNegotiating}`);
368     await this.remotePC.setRemoteDescription(desc);
369 
370     const answer = await this.remotePC.createAnswer();
371     this.debug('Create ANSWER: ', answer, this);
372     await this.remotePC.setLocalDescription(answer);
373     this.postData({
374       service     : _SVC.signaling.have_answer,
375       hub_id      : this.mget(_a.hub_id),
376       mode        : this.mget(_a.mode),
377       key         : Visitor.get(_a.socket_id),
378       socket_id   : this.mget(_a.socket_id),
379       answer
380     });
381     return answer;
382   }

388   async receiveAnswer(data) {
389     if(!this.isNegotiating) return;
390     this.debug("RECEIVED ANSWER 295", data);
391     const answer = data.answer;
392     const pc = this.localPC;
393     let desc = await new RTCSessionDescription(answer);
394 
395     this.debug(`XWQ TT SIG 372 == \
396     ${pc.signalingState} CONN=${pc.connectionState} NEG=${this.isNegotiating}`);
397     await this.remotePC.setLocalDescription(desc);
398     await this.localPC.setRemoteDescription(desc);
399   }

410     try{
411       await pc.addIceCandidate(desc);

413       this.warn("FAILED TO ADD CANDIDATE ", data);
414       this.debug(`XWQ ICE --> Gathering state  : ${pc.iceGatheringState}`, e);
415       this.debug(`XWQ ICE --> Connection state : ${pc.connectionState}`);
416       this.debug(`XWQ ICE --> Signaling state  : ${pc.signalingState}`);
417       this.warn("GOT ERROR", e);
418     }
419   }
