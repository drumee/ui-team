  7 let __singleton;
  8 const __STATIC_CLASSES__   = require('./seeds/static');
  9 const __DYNAMIC__CLASSES__ = require('./seeds/dynamic');
 10 const __USER_CLASSES__     = {};

 15 class __kind extends Backbone.Model {
 16   constructor(...args) {
 17     super(...args);
 18   }

 21 // ===========================================================
 22 //
 23 // ===========================================================
 24   exists(t) {
 25     return __STATIC_CLASSES__[t] || __USER_CLASSES__[t];
 26   }
 27 
 28 
 29 // ===========================================================
 30 // 
 31 // ===========================================================
 32   ensure(n, d) {
 33     if (this.exists(n)) {
 34       return this.get(n);
 35     }
 36     return this.register(n, d);
 37   }
 38 
 39 
 40 // ===========================================================
 41 //
 42 // ===========================================================
 43   register(k, v, verbose) {
 44     if (verbose == null) { verbose = 1; }
 45     if (__USER_CLASSES__[k]) {
 46       if (verbose) {
 47         this.warn(`Kind ${k} already exists. Use ->replace`);
 48       }
 49       return; 
 50     }
 51       //@warn "FORCE OVERWRITING KIND #{k}"
 52     if (__STATIC_CLASSES__[k]) {
 53       this.warn(`Kind *${k}* is static, it cannot be reused.`);
 54       return;
 55     }
 56     return __USER_CLASSES__[k] = v; 
 57   }
 58 
 59 // ===========================================================
 60 //
 61 // ===========================================================
 62   replace(k, v) {
 63     if (__STATIC_CLASSES__[k] || __DYNAMIC__CLASSES__[k]) {
 64       this.warn(`Kind ${k} is reserved. It cannot be resused`);
 65       return;
 66     }
 67     return __USER_CLASSES__[k] = v; 
 68   }
 69 
 70 // ===========================================================
 71 //
 72 // ===========================================================
 73   get(name) {
 74     let f = __STATIC_CLASSES__[name] || __USER_CLASSES__[name];
 75     if (_.isFunction(f)) {
 76       return f;
 77     }
 78     f = __DYNAMIC__CLASSES__[name];
 79     if (_.isFunction(f != null ? f.then : undefined)) {
 80       return require('./loader')(f);
 81     }
 82     this.warn(`Failed to fine kind for ${name}`);
 83     return null; 
 84   }

 87   // ===========================================================
 88   //
 89   // ===========================================================
 90   async waitFor (name){
 91     const self = this;
 92     let f = __STATIC_CLASSES__[name] || __USER_CLASSES__[name];
 93     if(_.isFunction(f)){
 94       return f;
 95     }
 96 
 97     f = __DYNAMIC__CLASSES__[name];
 98     if(!f){
 99       self.warn(`Could not wait for kind ${name}`);
100       return null;
101     }
102     if(_.isFunction(f.then)){
103       let r = await f.then(function(e){
104         self.register(name, e);
105         return e;
106       });
107       return r;
108     }
109   }

114   convert(letc, map, clone) {
115     let data, f;
116     if (clone == null) { clone = false; }
117     if (!letc) {
118       return null;
119     }
120     if (!(map != null ? map.match(/designer|reader/) : undefined)) {
121       _c.error("Type.convert requires map name");
122       return null;
123     }
124     if (clone) {
125       data = _.clone(letc);
126     } else {
127       data = letc;
128     }
129     if ((map != null ? map.match(/designer/) : undefined)) {
130       f = this.toDesigner;
131     } else {
132       f = this.toReader;
133     }
134     var walk = function(item){
135       //_dbg "AAAAA ---------> ", item.kind
136       let i;
137       item.kind = f(item.kind, map);
138       if (item.kids != null) {
139         for (i of Array.from(item.kids)) {
140           walk(i);
141         }
142       }
143       if (item.items != null) {
144         for (i of Array.from(item.items)) {
145           walk(i);
146         }
147       }
148       if (item.trigger != null) {
149         for (i of Array.from(item.trigger)) {
150           walk(i);
151         }
152       }
153       if (item.triggers != null) {
154         return (() => {
155           const result = [];
156           for (i of Array.from(item.triggers)) {
157             result.push(walk(i));
158           }
159           return result;
160         })();
161       }
162     };
163     if (_.isArray(data)) {
164       for (let d of Array.from(data)) {
165         if (d != null) {
166           walk(d);
167         }
168       }
169     } else {
170       if (data != null) {
171         walk(data);
172       }
173     }
174     data.kind = f(data.kind, map);
175     return data;
176   }

178 // ===========================================================
179 // splitFixed
180 // Because it's not possible to mix transform and fixed position,
181 // We have to split fixed_position object from the renderering tree
182 // @param [Object] letc
183 //
184 // @return [Object]
185 //
186 // ===========================================================
187   splitFixed(letc) {
188     const fixed  = [];
189     var walk = function(item){
190       if (item.fixed_position) {
191         fixed.push(_.clone(item));
192         item.kind = _t.wrapper;
193         item.className = "fixed-fixed"; 
194       }
195       //_dbg "AAAAA ---------> ", item.kind
196       if (item.kids != null) {
197         return Array.from(item.kids).map((i) =>
198           walk(i));
199       }
200     };
201     if (_.isArray(letc)) {
202       for (let d of Array.from(letc)) {
203         walk(d);
204       }
205     } else {
206       walk(letc);
207     }
208     const data = {
209       normal : letc,
210       fixed
211     };
212     return data;
213   }
