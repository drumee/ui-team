 87   /**
 88    * @param {(LetcBox)} cmd
 89    */
 90   saveOptions(cmd) {
 91     const option = cmd.parent
 92     const optionName = option.mget('formItem')
 93     const optionData = option.getData(_a.formItem)
 94     const commonData = {
 95       orgid: this.organisation.id,
 96       hub_id: Visitor.get(_a.id),
 97       option: optionData[optionName]
 98     }
 99     switch (optionName) {
100       case 'dir_visibility':
101         this.postData({
102           service: _SVC.adminpanel.organisation_update_dir_visiblity,
103           ...commonData
104         })
105         break;
106       case 'dir_info':
107         // _SVC.adminpanel.organisation_update_dir_info
108         this.postData({
109           service: _SVC.adminpanel.organisation_update_dir_info,
110           ...commonData
111         })
112         break;
113       default:
114         this.debug('undefined option')
115 
116     }
117 
118   }
