  7 /**
  8  * @typedef MenuItem
  9  * @type {object}
 10  * @property {string} id - Id OR Key for the menu item.
 11  * @property {string} display - Display item for the menu
 12  * @property {number} service - service on the selection 
 13  */
 14 
 15 /**
 16  * To create the Menu items 
 17  * @param {import('../index').initClass} a ___widget_dropdown_menu instance 
 18  * @param {MenuItem} opt - Options 
 19  */
 20 
 21 const __skl_dropdown_menu_item = function(_ui_,opt = {}) {
 22   const menuFig = `${_ui_.fig.family}-menu`;
 23 
 24   const menuItem = Skeletons.Box.X({
 25     className   : `${menuFig}__item`,
 26     service     : opt.service || 'change_option',
 27     kidsOpt     : {
 28       active      : 0
 29     },
 30     dataset     : { id: opt.id },
 31     kids        : [
 32       Skeletons.Note({
 33         className : `${menuFig}__note menu-item`,
 34         content   : opt.display || ''
 35       })
 36     ]});
 37     return menuItem;
 38 }
 39 /**
 40  * default Skeleton for dropdown_menu
 41  * @param {Object}  _ui_ 
 42  */
 43 const __skl_dropdown_menu = function(_ui_) {
 44 
 45   const menuFig = `${_ui_.fig.family}-menu`;
 46 
 47   const menuTrigger = Skeletons.Box.X({
 48     className : `${menuFig}__input_wrapper`,
 49     kids      : [
 50       Skeletons.Note({
 51         content: `Domain`,
 52         sys_pn : 'selected_text',
 53       }),
 54       Skeletons.Button.Svg({
 55         ico         : `editbox_arrow--down`,
 56         className   : `${menuFig}__icon-dropdown`,
 57       })
 58     ]
 59   })
 60 
 61   
 62   const options = _ui_.mget('options')
 63                     .map((option, index) => 
 64                       __skl_dropdown_menu_item(_ui_, { display: option.display, id: index }));
 65 
 66   const menuItems = Skeletons.Box.X({
 67     className : `${menuFig}__items-wrapper`,
 68     kids      : [
 69       Skeletons.Box.Y({
 70         className: `${menuFig}__items-wrapper fullwidth`,
 71         kids  : options
 72       })
 73     ]});
 74 
 75   const menu = Skeletons.Box.X({
 76     debug     : __filename,
 77     className : `${menuFig}__dropdown ${_ui_.fig.group}__dropdown`,
 78     kids      : [{
 79       kind        : _t.menu.topic,
 80       className   : `${menuFig}__wrapper ${_ui_.fig.group}__wrapper`,
 81       flow        : _a.y,
 82       opening     : _e.click,
 83       sys_pn      : "contact-card-dropdown",
 84       service     : "contact-card-menu",
 85       persistence : _a.none,
 86       trigger     : menuTrigger,
 87       items       : menuItems
 88     }]
 89   });
 90   
 91   return menu;
 92 };
 93 
 94 
 95 export default __skl_dropdown_menu;
