  7 __core = require('../core')
  8 class __endpoint_local extends __core {
  9   constructor(...args) {
 10     super(...args);
 11   }

 14   // ===========================================================

 16   // ===========================================================
 17   initialize(opt) {
 18     require('./skin');
 19     this.model.atLeast({
 20       mute:1 // To avoid Larsen
 21     });
 22     super.initialize(opt);
 23   }
 24   
 25   /* 
 26   * To inform others who is behind the particular localstream
 27   * 
 28   * ----------------------------------------------------------- */
 29   sendSignaling(data) {
 30     if(!this.stream || this.stream.id){
 31       return;
 32     }
 33     let opt = {
 34       service     : _SVC.signaling.message,
 35       event       : _e.update,
 36       stream_id   : this.stream.id,
 37       ...this.api,
 38       ...data
 39     };
 40     this.postData(opt);
 41   }
 42 
 43 
 44 
 45   // ===========================================================
 46   //
 47   // ===========================================================
 48   onDomRefresh() {
 49     this.start()
 50     .then((e)=>{
 51       this.setupTrack(_a.audio, this.mget(_a.audio));
 52       this.setupTrack(_a.video, this.mget(_a.video));
 53     })
 54     .catch((e)=>{
 55       this.triggerHandlers({
 56         status : _a.error,
 57         service : "devices-error",
 58         error: this.errorMessage
 59       });  
 60     });
 61   }
 62 
 63   /* --------------------------------------------------
 64    * Once access to user media (micro and/or video)
 65    * have been granted, pass stream provided by requestUserMedia to 
 66    * widget *webrtc_endpoint*
 67    * @params {object} {audio:boolean, video:boolean}
 68    * -------------------------------------------------- */
 69   async start() {
 70     this.devices = await this.probeDevices();
 71     this.stream = await this.requestUserMedia();
 72     this.triggerHandlers({
 73       devices:this.devices,
 74       stream:this.stream,
 75       localStream:this.stream,
 76       error: this.errorMessage
 77     });  
 78     this.feed(require('./skeleton')(this));
 79     this.ratio = this.$el.width() / this.$el.height(); 
 80     this.setupTrack(_a.video, this.mget(_a.video));
 81   }
 82 
 83   // ===========================================================
 84   //
 85   // ===========================================================
 86   async probeDevices() {
 87     const self =  this;
 88     //self.debug("SELECT JJJJJJJ");
 89     if (!navigator.mediaDevices) {
 90       this._failedToStart("No devices found");
 91       return;
 92     }
 93     let devices = null;
 94     try{
 95       devices = await navigator.mediaDevices.enumerateDevices();
 96     }catch(e){
 97       this._failedToStart(e);
 98     }
 99     return devices;
100   }
101 
102   // ===========================================================
103   //
104   // ===========================================================
105   _failedToStart(err){
106     this.warn("FAILED TO GET USER DEVICE", err);
107     window.__uu = err;
108     let msg = err;
109     if(err != null && err.message){
110       switch(err.name){
111         case "NotReadableError":
112           msg = `${err.message}. It's likely already in use.`;
113           break;
114         default:
115           msg = `${err.message}. Your browser doesn't support video feature.`;
116       }
117       if(this.__content){
118         this.__content.feed(
119           Skeletons.Note(msg, `${this.fig.family}__no-device`)
120         );
121       }else{
122         this.feed(
123           Skeletons.Note(msg, `${this.fig.family}__no-device`)
124         );  
125       }
126     }
127     this.errorMessage = msg;
128   };
129 
130   /* ------------------------------------------------------------
131   * To request permission to access user media (micro and/or video)
132   * @params {object} {audio:boolean, video:boolean}
133   * @retturn stream bound to granted media
134   * ------------------------------------------------------------ */
135   async requestUserMedia() {
136     let constraints = require('./configs');
137     // if(localStorage.deviceId){
138     //   constraints.video.deviceId = {
139     //     exact: localStorage.deviceId
140     //   }
141     // }
142     this.debug("XWQ requestUserMedia", constraints, this);
143     let  stream = null;
144     if (navigator.mediaDevices) {
145       stream = await navigator.mediaDevices.getUserMedia(constraints);
146     }
147     if(!stream){
148       this.warn("KKKKKKKKKKK")
149       this._failedToStart("No stream");
150     }
151     return stream;
152   } 
153 
154   /* ------------------------------------------------------------
155   * To request permission to access user media (micro and/or video)
156   * @params {object} {audio:boolean, video:boolean}
157   * @retturn stream bound to granted media
158   * ------------------------------------------------------------ */
159   async changeDevice(constraints) {
160     let  stream = null;
161     if (navigator.mediaDevices) {
162       stream = await navigator.mediaDevices.getUserMedia(constraints);
163     }
164     if(!stream){
165       this.warn("KKKKKKKKKKK")
166       this._failedToStart("No stream");
167     }
168     this__video.el.srcObject = stream;
169     this.stream = stream;
170     return stream;
171   }
