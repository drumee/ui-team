 10 function __skl_member_tags  (_ui_) {
 11   const tagsFig = _ui_.fig.family;
 12 
 13   const tag_id = _ui_.mget('radioId') || 'tag_selected'+_ui_.mget(_a.widgetId);
 14 
 15   const header = Skeletons.Box.X({
 16     className : `${tagsFig}__header`,
 17     kids      : [
 18       Skeletons.Note({
 19         className : `${tagsFig}__note title`,
 20         content   : LOCALE.TAG
 21       }),
 22 
 23       Skeletons.Button.Svg({
 24         ico       : 'desktop_plus',
 25         className : `${tagsFig}__icon add-tag desktop_plus`,
 26         service   : 'add-tag',
 27         uiHandler : _ui_
 28       })
 29     ]
 30   })
 31 
 32   const separator = Skeletons.Box.X({
 33     className : `${tagsFig}__separator`
 34   })
 35 
 36   const contentFig = `${tagsFig}-content`
 37   const tagList = Skeletons.List.Smart({
 38     className   : `${contentFig}__item list`,
 39     uiHandler   : [_ui_],
 40     service     : 'tag-list-data',
 41     flag        : 'all',
 42     timer       : 50,
 43     sys_pn      : _a.tags,
 44     api         : _ui_.getTags.bind(_ui_),
 45     itemsOpt    : {
 46       kind        : 'widget_member_tag_item',
 47       orgId       : _ui_.mget('orgId'),
 48       uiHandler   : [_ui_],
 49       active      : 0,
 50       radio       : tag_id
 51     }
 52   })
 53 
 54   const allMembers = Skeletons.Box.Y({
 55     className : `${contentFig}__main`,
 56     debug     : __filename,
 57     kids      : [
 58       Skeletons.Box.X({
 59         className : `${contentFig}__container widget-member-tagItem__ui`,
 60         service   : 'show-member-list',
 61         type      : 'allMembers',
 62         name      : 'All Members',
 63         sys_pn    : 'all-members',
 64         uiHandler : [_ui_],
 65         radio     : tag_id,
 66         dataset   : {
 67           state : 1,
 68           radio : _a.on
 69         },
 70         kidsOpt   : {
 71           active : 0
 72         },
 73         kids      : [
 74           Skeletons.Note({
 75             className : `${contentFig}__note name`,
 76             content   : 'All members'
 77           })
 78         ]
 79       })
 80     ]
 81   })
 82 
 83   const allAdmins = Skeletons.Box.Y({
 84     className : `${contentFig}__main`,
 85     debug     : __filename,
 86     kids      : [
 87       Skeletons.Box.X({
 88         className : `${contentFig}__container widget-member-tagItem__ui`,
 89         service   : 'show-member-list',
 90         type      : 'allAdmins',
 91         sys_pn    : 'all-admins',
 92         name      : 'All Administrators',
 93         uiHandler : [_ui_],
 94         radio     : tag_id,
 95         kidsOpt   : {
 96           active : 0
 97         },
 98         kids      : [
 99           Skeletons.Note({
100             className : `${contentFig}__note name`,
101             content   : 'All Administrators'
102           })
103         ]
104       })
105     ]
106   })
107 
108   const content = Skeletons.Box.X({
109     className : `${contentFig}`,
110     kids      : [
111       Skeletons.Box.Y({
112         className : `${contentFig}__items`,
113         kids      : [
114           Skeletons.Box.Y({
115             className : `${contentFig}__item`,
116             kids      : [
117               allMembers,
118               allAdmins,
119               tagList
120             ]
121           })
122         ]
123       })
124     ]
125   })
126 
127   let a = Skeletons.Box.Y({
128     className  : `${tagsFig}__main`,
129     debug      : __filename,
130     kids       : [
131       Skeletons.Box.Y({
132         className  : `${tagsFig}__container`,
133         kids : [
134           header,
135           separator,
136           content
137         ]
138       })
139     ]
140   })
141   
142   return a;
143 }
144 
145 export default __skl_member_tags;
