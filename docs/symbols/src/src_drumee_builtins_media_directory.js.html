  2 function traverseDirectory(entry, name) {
  3   const folder = entry.createReader();
  4   // Resolved when the entire directory is traversed
  5   if(!entry.isDirectory){
  6     console.warn("entry is not a directory");
  7     return 
  8 
  9   }
 10   let p = new Promise((resolve, reject) => {
 11     const queue = [name];
 12     function readEntries() {
 13       // According to the FileSystem API spec, readEntries() must be called until
 14       // it calls the callback with an empty array.  Seriously??
 15       folder.readEntries((entries) => {
 16         if (!entries.length) {
 17           // Done iterating this particular directory
 18           resolve(Promise.all(queue));
 19         } else {
 20           // Add a list of promises for each directory entry.  If the entry is itself
 21           // a directory, then that promise won't resolve until it is fully traversed.
 22           queue.push(Promise.all(entries.map((item) => {
 23             if (item.isFile) {
 24               return item;
 25             }
 26             return traverseDirectory(item, name);
 27           })));
 28           readEntries();
 29         }
 30       }, error => reject(error));
 31     }
 32     readEntries();
 33   });
 34   return p;
 35 }
 36 module.exports = traverseDirectory
