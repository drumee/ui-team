 71   /* ===========================================================
 72    *
 73    * ===========================================================*/
 74   onUiEvent (cmd, args){
 75     let service = cmd.get(_a.service) || cmd.get(_a.name);
 76     const status = cmd.get(_a.status);
 77     this.debug("onUiEvent",service,status, this)
 78 
 79     switch(service){
 80       case 'edit_domain':
 81         this.routePage('edit_domain')
 82         break;
 83       case 'view_domain':
 84         this.routePage('view_domain')
 85         break;
 86       case "added_domain":
 87         this.submitDomain();
 88         break;
 89       case "update_domain":
 90         this.submitDomain();
 91         break;
 92       case  _a.none:
 93         this.debug("Created by kind builder");
 94       default:
 95         this.debug("Created by kind builder");
 96     }
 97   }
 98 
 99   routePage(page){
100     switch(page){
101       case 'edit_domain':
102         this.mode = _a.edit
103         this.feed(require('./skeleton').default(this));
104         break;
105       case 'view_domain':
106         this.mode = _a.view
107         this.feed(require('./skeleton').default(this));
108         break;
109       default:
110         this.feed(require('./skeleton').default(this));
111     }
112   }
113   /* ***********************************************
114    *
115    * *********************************************** */
116   submitDomain(){
117     this.validateData()
118     if(this.formStatus == _a.error){
119       this.debug("invalid data")
120       return
121     }
122 
123     let data = this.getData(_a.formItem);
124     this.debug(data)
125     let service = _SVC.adminpanel.organisation_add;
126     if(this.mode == _a.edit){
127       service = _SVC.adminpanel.organisation_update;
128       data['orgid'] = this.organisation.id;
129     }
130     this.postData({
131       service : service,
132       hub_id  : Visitor.get(_a.id),
133       ...data
134     })
135   }

156   __dispatchRest(service, data, socket){
157     this.debug("Created by kind builder",service, data,this)
158 
159     switch(service){
160       case _SVC.adminpanel.organisation_add:
161       case _SVC.adminpanel.organisation_update:
162         if(data.status == 'IDENT_NOT_AVAILABLE'){
163           this.getPart('ident-entry').showError('Ooopsâ€¦this ident is already used. Try another one.')
164           return;
165         }
166         if(data.status == 'ORGANISATION_ALREADY_EXITS'){
167           this.warn('ORGANISATION_ALREADY_EXITS',data,this)
168         }
169         this.organisation = data;
170         this.mset('organisation',data);
171         this.routePage('view_domain')
172 
173         this.source   = this;
174         this.service  = 'domain_updated';
175         this.data     = data;
176         this.triggerHandlers();
177         this.service = '';
178     }
179   }
