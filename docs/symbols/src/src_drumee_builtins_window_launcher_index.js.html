 24   async onRender(opt) {
 25     if (opt == null) { opt = {}; }
 26     let api = this.mget(_a.api);
 27     let args  = _.clone(this.mget(_a.args));
 28     const hub_id = this.mget(_a.hub_id);
 29     this.debug("XXX launch  JJJJJJJJJJJ22", hub_id, args, this);
 30     if (api != null ? api.service : undefined) {
 31       this.fetchData(api);
 32     } else if (hub_id) {
 33       api = { 
 34         service : _SVC.media.attributes,
 35         nid     : this.mget(_a.nid) || '0',
 36         hub_id
 37       };
 38       if (['me', "*"].includes(hub_id)) {
 39         api.hub_id = Visitor.id;
 40       }
 41     } else { 
 42       args = _.merge(args, require('./applications')(args));
 43       //@debug "JJJJJJJJJJJ", args, require('./applications')(args)
 44       this.model.set(require('./applications')(args));
 45       if (args.kind === 'window_launcher') {
 46         this.warn(DEAD_LOCK);
 47         return;
 48       }
 49       Kind.waitFor(args.kind).then(this.renew.bind(this));
 50       return;
 51     }
 52     this.debug("XXX launch JJJJJJJJJJJapi", api.hub_id, api);
 53     const s = new Service(api);
 54     let data = await s.fetchData();
 55     this.run(data, args, opt);
 56     // if(data.error){
 57     //   this.failed(data);
 58     // }else{
 59     //   this.run(data, args, opt);
 60     // }
 61   }

 66   run(data, args, opt) {
 67     this.debug("XXX launch JJJJJJJJJJJ43", data, args, opt, this);
 68     this.model.clear();
 69     opt = _.merge(data, args);
 70     data = require('./applications')(opt);
 71     this.model.set(require('./applications')(opt));
 72     // @debug "JJJJJJJJJJJk=#{data.kind}", data, @
 73     if (data.kind === 'window_launcher') {
 74       this.warn(DEAD_LOCK);
 75       return;
 76     }
 77     Kind.waitFor(data.kind).then(this.renew.bind(this));
 78   }
