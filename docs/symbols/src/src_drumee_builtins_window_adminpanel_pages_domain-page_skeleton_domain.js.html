  7 function __skl_domain_form (_ui_){
  8   let org = _ui_.organisation
  9 
 10   if(_ui_.mode == _a.view){
 11     var nameInput = Skeletons.Note({
 12       className  : `${_ui_.fig.family}__input_display`,
 13       content: org.name
 14     })
 15   }else{
 16     const nameValidator = [      
 17       {reason: "Name is require" , comply: Validator.require}
 18     ]
 19     var nameInput = Skeletons.EntryBox({
 20       className   : `${_ui_.fig.family}__entry form-input`,
 21       formItem    : 'name',
 22       placeholder : LOCALE.ORGANISATION_NAME,//'Organisation name',
 23       value       : org.name || '',
 24       errorHandler  : [_ui_],
 25       validators    : nameValidator,
 26       showError   : true,
 27       sys_pn      : 'name-entry'
 28     })
 29   }
 30 
 31   const name = Skeletons.Box.X({
 32     className  : `${_ui_.fig.family}__form_row`,
 33     kids : [
 34       Skeletons.Note({
 35         className  : `${_ui_.fig.family}__row_label`,
 36         content: LOCALE.ORGA_NAME  // "Organisation name" + ":"
 37       }),
 38       Skeletons.Box.X({
 39         className  : `${_ui_.fig.family}__row_form_field`,
 40         kids: [
 41           nameInput
 42         ]
 43       })
 44     ]
 45   });
 46 
 47   if(_ui_.mode == _a.new){
 48     const identValidator = [      
 49       {reason: "Ident is require" , comply: Validator.require}, 
 50       {reason: "Enter a valid ident" , comply: Validator.ident}, 
 51     ]
 52     var identInput = Skeletons.EntryBox({
 53       className   : `${_ui_.fig.family}__entry form-input`,
 54       formItem    : 'ident',
 55       placeholder : LOCALE.IDENT, //'Ident',
 56       value       : '',
 57       errorHandler  : [_ui_],
 58       validators    : identValidator,
 59       showError   : true,
 60       sys_pn      : 'ident-entry'
 61     })
 62   }else{
 63     var identInput = Skeletons.Note({
 64       className  : `${_ui_.fig.family}__input_display`,
 65       content: org.ident
 66     })
 67 
 68   }
 69 
 70   
 71   
 72   const ident = Skeletons.Box.X({
 73     className  : `${_ui_.fig.family}__form_row`,
 74     kids : [
 75       Skeletons.Note({
 76         className  : `${_ui_.fig.family}__row_label`,
 77         content: LOCALE.ORGANISATION_IDENT //"Organisation Ident" + ":"
 78       }),
 79       Skeletons.Box.X({
 80         className  : `${_ui_.fig.family}__row_form_field`,
 81         kids: [
 82           identInput
 83         ]
 84       })
 85     ]
 86   });
 87   var linkText = ''
 88   if(_ui_.mode == _a.new){
 89     linkText = LOCALE.LINK_NOT_CREATED; //'Link has not been created yet';
 90   }else{
 91     linkText = org.link; 
 92   }
 93 
 94 
 95   const link = Skeletons.Box.X({
 96     className  : `${_ui_.fig.family}__form_row`,
 97     kids : [
 98       Skeletons.Note({
 99         className  : `${_ui_.fig.family}__row_label`,
100         content: LOCALE.ACCESS_LINK //"Access link" + ":"
101       }),
102       Skeletons.Box.X({
103         className  : `${_ui_.fig.family}__row_form_field`,
104         kids: [
105           Skeletons.Note({
106             className  : `${_ui_.fig.family}__link_display ${_ui_.mode}`,
107             content: linkText
108           })
109         ]
110       })
111     ]
112   });
113 
114 
115 
116   let a = Skeletons.Box.Y({
117     className  : `${_ui_.fig.family}__domain_form mode_${_ui_.mode}`,
118     debug      : __filename,
119     kids       : [
120       name,
121       ident,
122       link,
123       // buttons
124     ]
125   })
126   return a;
127 }
128 export default __skl_domain_form;
