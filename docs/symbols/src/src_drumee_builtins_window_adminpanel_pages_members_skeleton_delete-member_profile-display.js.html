  7 function __skl_member_delete_profile_display (_ui_) {
  8   const data = _ui_.mget(_a.member)
  9 
 10   const confirmProfileFig = `${_ui_.fig.family}-delete`;
 11 
 12   const firstname = data[_a.firstname] || ''
 13   const lastname  = data[_a.lastname] || ''
 14   const fullname  = data[_a.fullname] || (firstname + ' ' + lastname)
 15 
 16   const profile_icon = Skeletons.UserProfile({
 17     className : `${confirmProfileFig}__profile avatar`,
 18     id        : data['drumate_id'],
 19     firstname,
 20     lastname,
 21     fullname
 22   });
 23 
 24   const a = Skeletons.Box.X({
 25     className   : `${confirmProfileFig}__wrapper profile`,
 26     debug       : __filename,
 27     kids        : [
 28       profile_icon,
 29       
 30       Skeletons.Note({
 31         className : `${confirmProfileFig}__note fullname`,
 32         content   : fullname
 33       })
 34     ]
 35   });
 36 
 37   return a;
 38 };
 39 
 40 export default __skl_member_delete_profile_display;
