  7 const __skl_admin_panel_common_topBar = function(_ui_) {
  8   _dbg("__skl_admin_panel_common_topBar", _ui_);
  9 
 10   const mode      = _ui_._view;
 11   const figFamily = `${_ui_.fig.family}-topbar`;
 12   const figGroup  = `${_ui_.fig.group}-topbar`;
 13 
 14 
 15   let dropdownMenu  = ''
 16   let windowTitle = 'Administration';
 17   if (_ui_.organisation && _ui_.organisation.name) {
 18     windowTitle     = 'Administration' + ' - ' + _ui_.organisation.name
 19   }
 20 
 21   
 22   
 23   const a = Skeletons.Box.X({
 24     className : `${figFamily}__container ${figGroup}__container`,
 25     sys_pn    : _a.topBar,
 26     service   : _e.raise,
 27     dataset   : {
 28       view    : mode
 29     },
 30     debug     : __filename,
 31     kids      : [
 32       Skeletons.Box.X({
 33         className  : `${_ui_.fig.family}__page_dropdown`,
 34         sys_pn     :  'page_dropdown',
 35         kids: [dropdownMenu]
 36       }),
 37       Skeletons.Box.X({
 38         className : `${figFamily}__content ${figGroup}__content topbar-content`,
 39         kids      : [
 40           Skeletons.Note({
 41             sys_pn      : "window-name",
 42             uiHandler   : _ui_, 
 43             partHandler : _ui_,
 44             className   : "title",
 45             content     : windowTitle
 46           })
 47         ]}),
 48 
 49       require('window/skeleton/topbar/control')(_ui_, 'sc')
 50     ]});
 51 
 52   return a;
 53 };
 54 
 55 export default __skl_admin_panel_common_topBar;
