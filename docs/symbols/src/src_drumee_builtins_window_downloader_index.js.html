 39     let nodes = this.mget(_a.nodes);
 40     for(let node of nodes){
 41       node.unselect();
 42     }

 46 // ===========================================================

 49   onDomRefresh(){
 50     this.feed(require('./skeleton')(this));
 51     let nodes = this.mget(_a.nodes) || [];
 52 
 53     let list = [];
 54     let hub_id, nid;
 55     for(let node of nodes){
 56       hub_id = node.mget(_a.hub_id);
 57       if(node.mget(_a.filetype) == _a.hub) {
 58         nid = node.mget(_a.actual_home_id)
 59       }else{
 60         nid = node.mget(_a.nid);
 61       }
 62       list.push({hub_id, nid});
 63     }
 64     // Single folder
 65     if(this.mget(_a.folder)) list.push(this.mget(_a.folder));
 66     this._api = {
 67       service : _SVC.media.download,
 68       mode    : "dryrun",
 69       nodes   : list,
 70       nid     : 0,
 71       hub_id  : Visitor.id,
 72       socket_id : Visitor.get(_a.socket_id),
 73     };
 74     this.mset({
 75       hub_id : this._api.hub_id,
 76       nid : this._api.nid
 77     });
 78 
 79     this.fetchService(this._api, {async : 1}).then((data)=>{
 80       let content = LOCALE.TOTAL_SIZE_OF_FILES.format(Utils.humanFileSize(data.size));
 81       this.__filesize.set({content});
 82     });
 83     this.raise();
 84   }
 85

 87   // ===========================================================
 88   // 
 89   // ===========================================================
 90   downloadFiles() {
 91     this.started = 1;
 92     let nodes = this.mget(_a.nodes);    
 93     let f  = ()=>{
 94       var v = nodes.shift();
 95       if(!v){
 96         this.goodbye();
 97         return;
 98       }
 99       v.once(_e.loaded, f);
100       v.download();
101     }
102     f();
103   }

108   downloadZip() {
109     this.__method.set({content:""});
110     this.__wrapperStatus.feed(
111       Skeletons.Note(LOCALE.PROCESSING, "drumee-spinner status")
112     );
113     this._api.mode = ''; // wet run
114     // Since it's an archive, there is no specific filename
115     this._api.filename = moment().format("[drumee]-YYYY-MM-DD");
116     this.debug("zzzzzzzz", this._api);
117     //return;
118     this.fetchService(this._api, {async : 1}).then((data)=>{
119       let pn = `${this._id}-progres`;
120       if(!document.getElementById(pn)){
121         this.__wrapperStatus.feed({ 
122           kind : 'progress',
123           sys_pn   : pn,
124           filename : this._api.filename,
125           attributes : {
126             id : pn
127           } 
128         });  
129       }  
130     });
131   }

136   onUiEvent(cmd) {
137     const service = cmd.get(_a.service);
138     const status = cmd.get(_a.status);
139     this.debug(`SERVICE=${service}`);
140 
141     switch (service) {
142       case "download-files": case "download-zip":
143         this[_.camelCase(service)]();
144       break;
145 
146       default:
147         super.onUiEvent(cmd)
148     }
149   }

154   __dispatchPush(service, data){
155     switch (service) {
156       // Receive backend message saying archive is ready to be downloaded
157       case _SVC.media.download:
158         if(this.started) return;
159         if(!data.id){
160           this.debug("ZZZZZZZZ Compressing");
161           // let text = this.__wrapperStatus.children.last();
162           // text.set({content: "Compressing"});
163           return;
164         }
165         let url = Visitor.service({
166           service:_SVC.media.zip,
167           hub_id : this.mget(_a.hub_id),
168           nid    : this.mget(_a.nid),
169           id     : data.id, 
170           name   : data.name,
171           backup : this.mget(_a.backup)
172         });
173         let filename = `${data.name}.zip`;
174         let pn = `${this._id}-progres`;
175         this.started = 1;
176         Utils.waitForEl(pn, ()=>{
177           this.fetch({
178             url, 
179             progress : this.__wrapperStatus.children.last(), 
180             download : filename
181           }).then( ()=> {
182             _.delay(this.goodbye.bind(this), Visitor.timeout(1000));
183           })
184         })
185       }
186   }
