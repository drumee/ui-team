152   getLocalStream(id) {
153     let stream;
154     let first = this._streams.first();
155     if(id){
156       if(this._used.get(id)){
157         stream = first.clone();
158         this._streams.set(stream.id, stream);
159         return stream;
160       } 
161       stream = this._streams.get(id);
162       if(stream) return stream;
163     }
164     stream = first;
165     if(!this._used.get(stream.id)) return stream;
166     stream = first.clone();
167     this._streams.set(stream.id, stream);
168     return stream;
169   }

348   async requestUserMedia() {
349     const constraints = require('builtins/webrtc/endpoint/local/configs');
350     //this.debug("XWQ requestUserMedia", constraints, this);
351     let  stream = null;
352     if (navigator.mediaDevices) {
353       stream = await navigator.mediaDevices.getUserMedia(constraints);
354     }
355     if(!stream){
356       this._failedToStart("No stream");
357     }
358     return stream;
359   }

365   dispatchSignaling (data){
366     this.debug(`HHHHUUU1111=#${data.type}`, data, this);
367     let stream;
368     let label;
369     let s = data.sender;
370     let opt;
371     if(s) label = s.firstname || s.lastname ||s.email;
372     switch(data.type){
373       case _e.hello:
374         //stream = this._streams.last();
375         stream = this.getLocalStream();
376         opt = {
377           origin  : {
378             socket_id : Visitor.get(_a.socket_id),
379             stream_id : stream.id,
380           },
381           target  : {
382             socket_id : data.origin.socket_id,
383             stream_id : data.origin.stream_id,
384           },
385           scope : _a.socket,
386           type  : _e.handshake,
387         }
388         this.send(opt);
389         break;
390 
391       case _e.handshake:
392         stream = this.getLocalStream(data.target.stream_id);
393         this.debug(`HHHHUUU222=#${data.type}`, data, stream);
394         opt = {
395           origin  : {
396             socket_id : Visitor.get(_a.socket_id),
397             stream_id : stream.id,
398           },
399           target  : {
400             socket_id : data.origin.socket_id,
401             stream_id : data.origin.stream_id,
402           },
403           type : _e.connect,
404         }
405         this.send(opt);
406         opt.label       = label;
407         opt.service     = data.type,
408         opt.localStream = stream;
409         if(!this._fifo.length){
410           this.connect(opt);
411         }else{
412           this._fifo.push({opt});
413         }
414         break;
415       case _e.connect:
416         stream = this.getLocalStream(data.target.stream_id);
417         this.debug(`HHHHUUU333=#${data.type}`, data, stream);
418         opt = {
419           origin  : {
420             socket_id : Visitor.get(_a.socket_id),
421             stream_id : stream.id,
422           },
423           target  : {
424             socket_id : data.origin.socket_id,
425             stream_id : data.origin.stream_id,
426           },
427           remoteSettings:{
428             audio : data.audio,
429             video : data.video
430           }
431         }
432         opt.label       = label;
433         opt.service     = data.type,
434         opt.localStream = stream;
435         if(!this._fifo.length){
436           this.connect(opt);
437         }else{
438           this._fifo.push({opt});
439         }
440       break;
441       default:
442     }
443   }
