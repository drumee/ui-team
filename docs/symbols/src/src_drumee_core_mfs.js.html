 73     } else { 
 74       nid = this.mget(_a.nid);
 75     }
 76     this.postData({
 77       service   : _SVC.media.download,

 79       hub_id    : this.mget(_a.hub_id),

 89   getBlob(blob, filename) {
 90     const url = URL.createObjectURL(blob);
 91     const a = document.createElement(_K.tag.a);
 92     a.download = filename || 'download';
 93     a.hidden = '';
 94     a.setAttribute(_a.id, this._id + '-dl');
 95     a.setAttribute(_a.href, url);
 96     a.setAttribute(_a.target, '_blank');
 97     a.setAttribute("data-service", _e.download);
 98     a.style.position = _a.absolute;
 99     a.style.display = _a.none;
100     this.unselect();
101     var clickHandler = ()=> {
102       const f = ()=> {
103         URL.revokeObjectURL(url);
104         a.removeEventListener(_e.click, clickHandler);
105         this.trigger(_e.eod, blob);
106       };
107       _.delay(f, 300);
108     };
109     a.addEventListener(_e.click, clickHandler, false);
110     a.click();
111   }

116   async fetch(o) {
117     //let url, progress, filename
118     // const self = this;
119     return fetch(o.url).then((response) => {
120       const total = Number(response.headers.get('content-length'));
121       var reader = response.body.getReader();
122       var loaded = 0;
123       let type = response.headers.get('content-type').split(' ')[0];
124       const buffer = new ArrayBuffer(total);
125       let content = new Uint8Array(buffer);
126 
127       return reader.read().then(function read_data(result) {
128         if (result.done) {
129           //console.log('Fetch complete', result, reader, content);
130           if(o.progress) o.progress.goodbye();
131           reader.releaseLock();
132           return new Blob([content], {type});
133         }
134 
135         // enqueue received data
136         content.set(result.value, loaded);
137         // result.value for fetch streams is a Uint8Array
138         loaded += result.value.length;
139         if(o.progress){
140           o.progress.update({loaded, total});  
141         }else{
142           this.trigger(_e.progress, {loaded, total});
143         }
144         return reader.read().then(read_data);
145 
146       }).then((blob)=>{
147         if(o.download){
148           this.getBlob(blob, o.download);
149         }else{
150           this.trigger(_e.eod, blob);
151           return blob;
152         }
153       });
154     });
155   }

160   download(o) {
161     let nid;
162     let url = this.url(_a.orig);
163     let type = this.mget(_a.filetype);
164     let filename = this.mget(_a.filename);
165     filename = filename.replace(/\<.+\>/, '');
166     if (type === _a.hub) { 
167       nid = this.mget(_a.actual_home_id);
168     } else { 
169       nid = this.mget(_a.nid);
170     }
171     let pn = `${this._id}-progres`
172     if(!document.getElementById(pn)){
173       this.append({ 
174         kind : 'progress',
175         sys_pn   : pn,
176         filename : filename,
177         attributes : {
178           id : pn
179         } 
180       });  
181     }
182     this.trigger(_e.loaded);
183     if ([_a.hub, _a.folder].includes(type)) {
184       if (_.isEmpty(o)) {
185         this.download_tree();
186         return;
187       }
188       url = Visitor.service({
189         service:_SVC.media.zip,
190         hub_id : this.mget(_a.hub_id),
191         nid,
192         id     : o.id, 
193         name   : o.name
194       });
195       filename = `${filename}.zip`;
196     }else{
197       filename = `${filename}.${this.mget(_a.extension)}`;
198     }
199     Utils.waitForEl(pn, ()=>{
200       this.fetch({
201         url, 
202         progress : this.children.last(), 
203         download:filename
204       });
205     })
206   }

265       hub_id   : m.get(_a.hub_id)
