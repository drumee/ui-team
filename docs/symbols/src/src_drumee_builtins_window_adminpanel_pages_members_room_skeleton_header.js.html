 10 function __skl_members_room_header (_ui_) {
 11   const headerFig = `${_ui_.fig.family}-header`;
 12   
 13   let titleContent, titleIcon, actionIcon, drumateId, roleAutoSave, roleFetch;
 14   
 15   if (_ui_.mget(_a.type) == 'member_create') {
 16     titleContent = 'Create a member'
 17     roleAutoSave = 0
 18     roleFetch = 0
 19 
 20   } else {
 21     const data = _ui_.mget(_a.member) || _ui_._currentData.model.attributes || ''
 22 
 23     const fname = data.firstname  || '';
 24     const lname = data.lastname  || '';
 25     const fullname = data.fullname || (fname  + " " + lname);
 26     titleContent = fullname
 27     drumateId = data.drumate_id
 28 
 29     titleIcon = Skeletons.UserProfile({
 30       className : `${headerFig}__profile`,
 31       id        : drumateId,
 32       firstname : fname,
 33       lastname  : lname,
 34       fullname  : fullname
 35     });
 36 
 37     if (_ui_._type == 'member_detail') {
 38       actionIcon = Skeletons.Box.X({
 39         className   : `${headerFig}__wrapper options-right`,
 40         kids        : [
 41           Skeletons.Button.Svg({
 42             ico       : 'desktop_sharebox_edit',
 43             className : `${headerFig}__icon edit-icon desktop_sharebox_edit`,
 44             service   : _e.edit,
 45             uiHandler : _ui_ 
 46           })
 47         ]
 48       });
 49     }
 50     ////////////////////// POSTPONED - will be implemented after introduced Member Block feature //////////////////////
 51     // } else if (_ui_._type == 'member_edit') {
 52     //   actionIcon = Skeletons.Box.X({
 53     //     className   : `${headerFig}__wrapper options-right`,
 54     //     kids        : [
 55     //       Skeletons.Button.Svg({
 56     //         ico       : 'tools_delete',
 57     //         className : `${headerFig}__icon delete-icon tools_delete`,
 58     //         service   : _e.delete,
 59     //         uiHandler : _ui_
 60     //       })
 61     //     ]
 62     //   });
 63     // }
 64   
 65   }
 66  
 67   const title = Skeletons.Box.X({
 68     className   : `${headerFig}__title`,
 69     kids        : [
 70       titleIcon,
 71 
 72       Skeletons.Note({
 73         className   : `${headerFig}__note title`,
 74         content     : titleContent
 75       })
 76     ]
 77   });
 78 
 79   if (_ui_._type == 'member_edit') {
 80     roleAutoSave = 0
 81     roleFetch = 1
 82   }
 83 
 84   const roleMenu = {
 85     kind       : 'widget_member_roles_menu',
 86     sys_pn     : 'member-role-menu',
 87     className  : `${headerFig}__wrapper options-left`,
 88     userId     : drumateId,
 89     orgId      : _ui_.mget('orgId'),
 90     autoSave   : roleAutoSave,
 91     fetchRoles : roleFetch
 92   }
 93   
 94   const a = Skeletons.Box.X({
 95     className   : headerFig,
 96     debug      : __filename,
 97     sys_pn      : 'members_room_header',
 98     kids        : [
 99       roleMenu,
100       title,
101       actionIcon
102     ]
103   })
104   
105   return a;
106 };
107 
108 export default __skl_members_room_header;
