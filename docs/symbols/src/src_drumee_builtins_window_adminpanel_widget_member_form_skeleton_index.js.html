 10 function __skl_widget_member_form  (_ui_) {
 11   const formFig = _ui_.fig.family
 12   const data = _ui_._memberData
 13 
 14   const emailValidators = [      
 15     {reason: LOCALE.ENTER_VALID_EMAIL , comply: Validator.email},
 16     {reason: LOCALE.EMAIL_REQUIRED , comply: Validator.require}
 17   ]
 18   
 19   const name = Skeletons.Box.X({
 20     className  : `${formFig}__wrapper name`,
 21     kids       : [
 22       Skeletons.Button.Svg({
 23         ico         : 'profile-signup',
 24         className   : `${formFig}__icon name profile-signup`
 25       }),
 26     
 27       Skeletons.Box.Y({
 28         className   : `${formFig}__entry-wrapper name`,
 29         kids        : [
 30           Skeletons.EntryBox({
 31             className   : `${formFig}__entry firstname`,
 32             formItem    : _a.firstname,
 33             placeholder : LOCALE.FIRSTNAME,
 34             value       : data.firstname
 35           }),
 36 
 37           Skeletons.Entry({
 38             className   : `${formFig}__entry familyname`,
 39             formItem    : _a.lastname,
 40             placeholder : LOCALE.LASTNAME,
 41             value       : data.lastname
 42           })
 43         ]
 44       })
 45     ]
 46   })
 47 
 48   const email = Skeletons.Box.X({
 49     className   : `${formFig}__wrapper email`,
 50     kids        : [
 51       Skeletons.Button.Svg({
 52         ico         : 'email',
 53         className   : `${formFig}__icon email`
 54       }),
 55 
 56       Skeletons.EntryBox({
 57         className   : `${formFig}__entry email`,
 58         formItem    : _a.email,
 59         innerClass  : _a.email,
 60         interactive : 1,
 61         sys_pn      : 'email-input',
 62         placeholder : LOCALE.EMAIL,
 63         value       : data.email,
 64         uiHandler     : _ui_,
 65         errorHandler  : [_ui_],
 66         validators    : emailValidators,
 67         showError     : 1
 68       }),
 69     ]
 70   })
 71 
 72   const ident = Skeletons.Box.X({
 73     className   : `${formFig}__wrapper ident`,
 74     kids        : [
 75       Skeletons.Button.Svg({
 76         ico         : 'account_id',
 77         className   : `${formFig}__icon ident account_id`
 78       }),
 79 
 80       Skeletons.EntryBox({
 81         className     : `${formFig}__entry ident`,
 82         formItem      : _a.ident,
 83         innerClass    : _a.ident,
 84         interactive   : 1,
 85         sys_pn        : 'ident-input',
 86         placeholder   : 'Ident',
 87         value         : data.ident,
 88         uiHandler     : _ui_,
 89         errorHandler  : [_ui_],
 90         validators    : [
 91           { reason: 'Ident Required' , comply: Validator.require }
 92         ],
 93         showError     : 1
 94       }),
 95     ]
 96   })
 97 
 98   const mobile = Skeletons.Box.X({
 99     className   : `${formFig}__wrapper mobile`,
100     kids        : [
101       Skeletons.Button.Svg({
102         ico         : 'account_phone',
103         className   : `${formFig}__icon mobile account_phone`
104       }),
105 
106       Skeletons.EntryBox({
107         className     : `${formFig}__entry mobile`,
108         formItem      : _a.mobile,
109         innerClass    : _a.mobile,
110         interactive   : 1,
111         placeholder   : LOCALE.PHONE,
112         value         : data.mobile,
113         uiHandler     : _ui_,
114         errorHandler  : [_ui_],
115         showError     : 1,
116         validators    : [
117           { reason: LOCALE.VALID_PHONE_NO , comply: Validator.phone }
118         ],
119       }),
120     ]
121   })
122 
123   const address = Skeletons.Box.X({
124     className   : `${formFig}__wrapper address`,
125     kids        : [
126       Skeletons.Button.Svg({
127         ico         : 'ab_address',
128         className   : `${formFig}__icon address ab_address`
129       }),
130 
131       Skeletons.Box.Y({
132         className : `${formFig}__entry-wrapper address`,
133         formItem  : _a.address,
134         dataType  : _a.object,
135         kids      : [
136           Skeletons.Entry({
137             className   : `${formFig}__entry street address-item`,
138             formItem    : _a.street,
139             placeholder : LOCALE.NUMBER_STREET,
140             uiHandler   : _ui_,
141             value       : data.address.street
142           }),
143         
144           Skeletons.Entry({
145             className   : `${formFig}__entry zip address-item`,
146             formItem    : _a.city,
147             placeholder : LOCALE.ZIP_CODE_CITY,
148             uiHandler   : _ui_,
149             value       : data.address.city
150           }),
151           
152           Skeletons.Entry({
153             className   : `${formFig}__entry country address-item`,
154             formItem    : _a.country,
155             placeholder : LOCALE.COUNTRY,
156             uiHandler   : _ui_,
157             value       :data.address.country
158           })
159         ]
160       })
161     ]
162   })
163 
164   const comment = Skeletons.Box.X({
165     className   : `${formFig}__wrapper comment`,
166     kids        : [
167       Skeletons.Box.X({
168         className   : `${formFig}__entry-wrapper comment`,
169         kids        : [
170           Skeletons.Entry({
171             className   : `${formFig}__entry comment`,
172             type        : _a.textarea,
173             formItem    : _a.comment,
174             placeholder : LOCALE.NOTES,
175             value       : data.comment|| ''
176           })
177         ]
178       })
179     ]
180   })
181 
182   let buttonLabel = LOCALE.ADD
183   if (_ui_._type == 'member_edit')
184     buttonLabel = LOCALE.MODIFY
185   
186   const buttons = Preset.ConfirmButtons(_ui_, 
187     { 
188       confirmLabel  : buttonLabel , 
189       cancelService : _e.cancel
190     },
191     {
192       sys_pn  : 'submit-button'
193     })
194 
195   let a = Skeletons.Box.Y({
196     className  : `${formFig}__main`,
197     debug      : __filename,
198     kids       : [
199       Skeletons.Box.Y({
200         className  : `${formFig}__container`,
201         kids : [
202           name,
203           email,
204           ident,
205           mobile,
206           address,
207           comment,
208           buttons
209         ]
210       })
211     ]
212   })
213 
214   return a;
215 }
216 
217 export default __skl_widget_member_form
