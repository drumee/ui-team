  7 function __skl_member_delete_confirmation (_ui_) {
  8   const confirmFig = `${_ui_.fig.family}-delete`;
  9 
 10   const close = Skeletons.Box.X({
 11     className   : `${confirmFig}__close`,
 12     kids        : [
 13       Skeletons.Button.Svg({
 14         ico         : 'account_cross',
 15         className   : `${confirmFig}__icon close account_cross`,
 16         service     : 'close-overlay',
 17         uiHandler   : _ui_
 18       })
 19     ]
 20   });
 21   
 22   const header = Skeletons.Note({
 23     className :  `${confirmFig}__note header`,
 24     content: LOCALE.DELETE_CONTACT
 25   });
 26 
 27   const subTitle = Skeletons.Note({
 28     className :  `${confirmFig}__note sub-title`,
 29     content: LOCALE.CONFIRM_DELETE_CONTACT
 30   });
 31 
 32   const profileDisplay = require('./profile-display').default(_ui_);
 33   
 34   const buttons = Preset.ConfirmButtons(_ui_, {
 35     cancelLabel     : LOCALE.CANCEL || '',
 36     cancelService   : 'close-overlay',
 37     confirmLabel    : LOCALE.DELETE || '',
 38     confirmService  : 'confirm-delete'
 39   });
 40 
 41   const bodyContent = Skeletons.Box.Y({
 42     className   : `${confirmFig}__content`,
 43     sys_pn      : 'delete-confirm-content',
 44     kids        : [
 45       header,
 46       subTitle,
 47       profileDisplay,
 48       buttons
 49     ]
 50   });
 51   
 52   const a = Skeletons.Box.X({
 53     debug       : __filename,
 54     className   : `${confirmFig}__confirmation`,
 55     kids        : [
 56       close,
 57       bodyContent
 58     ]
 59   });
 60   
 61   return a;
 62 };
 63 
 64 export default __skl_member_delete_confirmation;
